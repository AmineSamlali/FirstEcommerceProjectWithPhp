    function checkUserStatus( $username = null , $password = null,$userId = null, $redirectMode=false ){
        global $conn;
        if($username and $password){
            $connection = $conn->prepare("SELECT reg_status FROM shop.users WHERE username = ? AND password = ?");
            $connection->execute([$username,$password]);
            $data = $connection->fetchAll();
            if($connection->rowCount()){
                if($redirectMode){
                    if($data['reg_status']){
                        header('location:logout.php');
                        exit();
                    }else{
                        return $data;
                    }
                }
            }
            return 1;
        }
